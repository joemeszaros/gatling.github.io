class MySimulation extends Simulation {

  val conf = http.baseUrl("http://localhost")

  val scn = scenario("Gatling")
      .exec(http("index").get("/"))
      .during(10 minutes) {
    exec(
      http("json").get("/json")
        .check(jsonPath("$.id")
        .saveAs("id"))
    )
  }

  setUp(scn.inject(atOnceUsers(5)))
        .protocols(conf)
}

Linux : gatling.sh
Windows : gatling.bat

Maven : mvn gatling:execute
SBT : sbt test
